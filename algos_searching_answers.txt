function binarySearch(binaryArray, x) {
  var low = 0;
  var high = binaryArray.length - 1;
  
  while(low <= high) {
    var mid = Math.floor(low + high % 2);
    if(binaryArray[mid] > x) {
      high = mid - 1;
    } else if(binaryArray[mid] < x) {
      low = mid + 1;
    } else {
      return binaryArray[mid];
    }
  }
}
